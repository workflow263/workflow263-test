JS实现为控件添加倒计时功能

1. 描述

在有些报表需求中，需要为控件添加倒计时功能，限制到某一个时间点后能进行一项操作或不能进行某项操作，比如查询，导出功能等等，又需要人性化地显示还有多少时间，即倒计时功能，比如下图中我们限制这个报表在每天17点50分后才能查询：

2. 思路

主要原理是利用控件的setEnable(true)/setEnable(false)来进行设置控件的可用与不可用状态，在获取时间的过程中，需要利用到JS中的获取时间，利用JS的定时器函数setInterval(function(){},time)来进行定时取得倒时时，并判断倒计时是否结束。

3.2 添加倒计时控制功能

为了简化控制流程，把JS代码直接写在查询按钮的初始化后事件中，如下图：

  1. var h = 17; //限制几点可查询  
  2. var m = 50; //限制几分可查询  
  3. var s = 00; //限制几秒可查询  
  4.   
  5. //格式化时间  
  6. function timeToString(a) {  
  7.     //小时  
  8.     var s = '还有';  
  9.     s += parseInt(a / 3600) + '时';  
  10.     //分  
  11.     s += parseInt(a % 3600 / 60) + '分';  
  12.     //秒  
  13.     s += parseInt(a % 60) + '秒可查';  
  14.     return s;  
  15. }  
  16.   
  17. var date1 = new Date();  
  18. var date2 = new Date();  
  19. //设置预置可查时间  
  20. date1.setHours(h);  
  21. date1.setMinutes(m);  
  22. date1.setSeconds(s);  
  23.   
  24. //比如时间  
  25. var d = (date1 - date2) / 1000;  
  26. //如果初始化时可用，就启用按钮  
  27. if(d < 0) {  
  28.     this.setValue('查询');  
  29.     this.setEnable(true);  
  30. } else {  
  31.     var btn = this;  
  32.     //显示倒计时时间  
  33.     btn.setValue(timeToString(d));  
  34.     //设置不可用  
  35.     btn.setEnable(false);  
  36.     //定时器函数  
  37.     setInterval(function() {  
  38.         //重新设置时间  
  39.         date1 = new Date();  
  40.         date2 = new Date();  
  41.         date1.setHours(h);  
  42.         date1.setMinutes(m);  
  43.         date1.setSeconds(s);  
  44.         //重新当前时间与设定时间的时间差  
  45.         d = (date1 - date2) / 1000;  
  46.         if(d < 0) {  
  47.             btn.setValue('查询');  
  48.             btn.setEnable(true);  
  49.         } else {  
  50.             btn.setValue(timeToString(d));  
  51.             btn.setEnable(false);  
  52.         }  
  53.     }, 1000);  
  54. }
